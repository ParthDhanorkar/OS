FCFS
#!/bin/bash
echo "Enter number of processes:"
read n
for (( i=0; i<n; i++ ))
do
    echo "Enter Burst Time for Process $((i+1)):"
    read bt[$i]
done
wt[0]=0
for (( i=1; i<n; i++ ))
do
    wt[$i]=$((wt[i-1] + bt[i-1]))
done
echo "Process   BT   WT   TAT"
for (( i=0; i<n; i++ ))
do
    tat[$i]=$((bt[i] + wt[i]))
    echo "P$((i+1))        ${bt[$i]}    ${wt[$i]}    ${tat[$i]}"
done


touch fcfs.sh
vim fcfs.sh
chmod +x fcfs.sh
./fcfs.sh
