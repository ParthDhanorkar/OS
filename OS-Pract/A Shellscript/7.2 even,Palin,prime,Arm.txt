#!/bin/bash
echo "Enter number:"
read n
echo "1.Even/Odd"
echo "2.Prime"
echo "3.Palindrome"
echo "4.Armstrong"
echo "Enter choice:"
read ch


case $ch in
    1)
        if [ $((n%2)) -eq 0 ]
        then
            echo "Even Number"
        else
            echo "Odd Number"
        fi
        ;;
    2)
        flag=0
        for (( i=2; i<=n/2; i++ ))
        do
            if [ $((n%i)) -eq 0 ]
            then
                flag=1
                break
            fi
        done
        if [ $flag -eq 0 ]
        then
            echo "Prime Number"
        else
            echo "Not Prime"
        fi
        ;;
    3)
        rev=$(echo $n | rev)
        if [ $n -eq $rev ]
        then
            echo "Palindrome Number"
        else
            echo "Not Palindrome"
        fi
        ;;
    4)
        sum=0
        temp=$n
        while [ $temp -gt 0 ]
        do
            rem=$((temp % 10))
            sum=$((sum + rem*rem*rem))
            temp=$((temp / 10))
        done
        if [ $n -eq $sum ]
        then
            echo "Armstrong Number"
        else
            echo "Not Armstrong"
        fi
        ;;
    *)
        echo "Invalid Choice"
        ;;
esac


nano num_check.sh
chmod +x num_check.sh
./num_check.sh
