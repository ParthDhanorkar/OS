#!/bin/bash
echo "How many items in the list?"
read n
echo "Enter $n items:"


# Loop to take input
for (( i=1; i<=n; i++ ))
do
    read item
    list="$list $item"
done


echo "Your list is: $list"


# Example: Search in the list
echo "Enter element to search:"
read e
found=0
for i in $list
do
    if [ $i -eq $e ]
    then
        found=1
        break
    fi
done


if [ $found -eq 1 ]
then
    echo "Element Found"
else
    echo "Element Not Found"
fi
